    <script>
        let currentDeg = 0;

        function spin() {
            const wheel = document.getElementById('wheel');
            const resultText = document.getElementById('result');
            
            // Reset kết quả
            resultText.innerText = "Đang quay...";

            // Tạo số vòng quay ngẫu nhiên (ít nhất 5 vòng + phần lẻ ngẫu nhiên)
            // 360 * 5 = 1800 độ tối thiểu
            const randomDeg = Math.floor(Math.random() * 360) + 1800; 
            
            // Cộng dồn vào góc quay hiện tại để bánh xe không bị quay ngược lại
            currentDeg += randomDeg;

            wheel.style.transform = `rotate(${currentDeg}deg)`;

            // Tính toán kết quả sau khi quay xong (4 giây)
            setTimeout(() => {
                // Tính góc thực tế (chia lấy dư cho 360)
const actualDeg = currentDeg % 360;
                
                // Lưu ý: Góc 0 độ ở CSS conic-gradient bắt đầu từ hướng 12h và chạy theo chiều kim đồng hồ.
                // Kim chỉ ở hướng 12h.
                // Do đó, phần thưởng trúng là phần nằm dưới kim chỉ.
                // Cần tính toán ngược lại một chút tùy vào logic chia màu của bạn.
                
                resultText.innerText = "Đã dừng lại! Chúc mừng!";
            }, 4000);
        }
    </script>
